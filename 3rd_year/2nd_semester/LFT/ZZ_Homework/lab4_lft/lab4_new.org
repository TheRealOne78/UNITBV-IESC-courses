** Implementati acest AFD in Rust adaugand un nou enum OpState si struct OpDfa in src/dfa.rs, urmand acelasi pattern ca IdentDfa.

*** ~dfa.rs~

#+begin_src rust
#[derive(Debug, Clone, PartialEq)]
pub enum OpState {
    Start, SawEq, SawEqEq, SawBang, SawNotEq, SawMinus, SawArrow, Dead,
}

pub struct OpDfa;

impl Dfa for OpDfa {
    type State = OpState;

    fn start() -> OpState { OpState::Start }

    fn transition(state: &OpState, c: char) -> OpState {
        match state {
            OpState::Start    => match c {
                '=' => OpState::SawEq,
                '!' => OpState::SawBang,
                '-' => OpState::SawMinus,
                _   => OpState::Dead,
            },
            OpState::SawEq    => if c == '=' { OpState::SawEqEq  } else { OpState::Dead },
            OpState::SawBang  => if c == '=' { OpState::SawNotEq } else { OpState::Dead },
            OpState::SawMinus => if c == '>' { OpState::SawArrow } else { OpState::Dead },
            _                 => OpState::Dead,
        }
    }

    fn is_accepting(state: &OpState) -> bool {
        matches!(state,
            OpState::SawEq | OpState::SawEqEq | OpState::SawNotEq |
            OpState::SawMinus | OpState::SawArrow)
    }
}
#+end_src

pub enum NumberState {

* Exercitiul 5: Minimizarea manuala a unui AFD

** Considerati AFD-ul cu 4 stari {A, B, C, D} peste alfabetul {0, 1}, cu starile finale {C, D}, si tranzitiile: A-0->B, A-1->C, B-0->B, B-1->D, C-0->C, C-1->C, D-0->D, D-1->D.
